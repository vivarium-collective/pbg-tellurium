Composite spec (YAML)

name: decay-with-viz
description: "Tellurium decay demo with a wired Visualization Step. Each update produces a Plotly HTML figure of the running species trajectories, written to the 'viz_html' store."

requires:
  processes: [TelluriumProcess, SpeciesTimeSeriesPlots, RAMEmitter]

parameters:
  interval:
    type: float
    default: 1.0
    description: "Process interval in model time units."
  viz_title:
    type: string
    default: "Tellurium decay (S1 -> S2)"
    description: "Title displayed at the top of the Plotly figure."

state:
  tellurium:
    _type: process
    address: "local:TelluriumProcess"
    config:
      model: |
        model decay
          S1 = 10; S2 = 0
          S1 -> S2; k*S1; k = 0.3
        end
      model_format: "antimony"
      integrator: "cvode"
    inputs: {}
    outputs:
      species: ["stores", "species"]
      rates: ["stores", "rates"]
      parameters: ["stores", "parameters"]
      time: ["stores", "time"]
    interval: "${interval}"
  stores:
    species: {}
    rates: {}
    parameters: {}
    time: 0.0
    viz_html: ""
  viz:
    _type: step
    address: "local:SpeciesTimeSeriesPlots"
    config:
      title: "${viz_title}"
    inputs:
      species: ["stores", "species"]
      time: ["stores", "time"]
    outputs:
      html: ["stores", "viz_html"]
  emitter:
    _type: step
    address: "local:RAMEmitter"
    config:
      emit:
        species: "map[float]"
        rates: "map[float]"
        time: "float"
    inputs:
      species: ["stores", "species"]
      rates: ["stores", "rates"]
      time: ["stores", "time"]
