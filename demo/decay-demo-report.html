Composite spec (YAML)

name: decay-demo
description: "Tiny Tellurium / Antimony decay demo: S1 -> S2 first-order kinetics. Emits species concentrations, reaction rates, and time each step."

requires:
  processes: [TelluriumProcess, RAMEmitter]

parameters:
  interval:
    type: float
    default: 1.0
    description: "Process interval in model time units. Each update advances RoadRunner by this much."
  k_decay:
    type: float
    default: 0.3
    description: "First-order decay rate constant for S1 -> S2."
  s1_initial:
    type: float
    default: 10.0
    description: "Initial concentration of S1."

state:
  tellurium:
    _type: process
    address: "local:TelluriumProcess"
    config:
      model: |
        model decay
          S1 = 10; S2 = 0
          S1 -> S2; k*S1; k = 0.3
        end
      model_format: "antimony"
      integrator: "cvode"
      species_overrides:
        S1: "${s1_initial}"
      parameter_overrides:
        k: "${k_decay}"
    inputs: {}
    outputs:
      species: ["stores", "species"]
      rates: ["stores", "rates"]
      parameters: ["stores", "parameters"]
      time: ["stores", "time"]
    interval: "${interval}"
  stores:
    species: {}
    rates: {}
    parameters: {}
    time: 0.0
  emitter:
    _type: step
    address: "local:RAMEmitter"
    config:
      emit:
        species: "map[float]"
        rates: "map[float]"
        time: "float"
    inputs:
      species: ["stores", "species"]
      rates: ["stores", "rates"]
      time: ["stores", "time"]
